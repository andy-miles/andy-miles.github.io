package com.amilesend.discogs.parse.creator;

import com.amilesend.discogs.connection.DiscogsConnection;
import com.amilesend.discogs.model.marketplace.GetInventoryResponse;
import com.google.gson.InstanceCreator;
import lombok.RequiredArgsConstructor;

import java.lang.reflect.Type;

/**
 * A custom {@link InstanceCreator} implementation that injects the {@link DiscogsConnection} to the
 * resource type so that method operations can be performed on the {@link GetInventoryResponse}
 * resource.
 *
 * @see InstanceCreator
 * @see GetInventoryResponse
 */
@RequiredArgsConstructor
public class GetInventoryResponseCreator implements InstanceCreator<GetInventoryResponse> {
    /** The current client connection instance. */
    private final DiscogsConnection connection;

    @Override
    public GetInventoryResponse createInstance(final Type type) {
        return GetInventoryResponse.builder().connection(connection).build();
    }
}
