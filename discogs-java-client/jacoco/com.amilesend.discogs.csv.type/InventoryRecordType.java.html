/**
 * Describes the record type that is either used for creating a new inventory item, or update an existing one.
 *
 * @see InventoryRecord
 */
public enum
InventoryRecordType {
    /** The record is for a new inventory item. */
    NEW,
    /** The record is to update an existing inventory item. */
    UPDATE;
}
