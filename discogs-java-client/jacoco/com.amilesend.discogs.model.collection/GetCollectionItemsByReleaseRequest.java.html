package com.amilesend.discogs.model.collection;

import com.amilesend.client.util.Validate;
import com.amilesend.discogs.model.PaginatedRequestBase;
import com.amilesend.discogs.model.PathParameter;
import lombok.EqualsAndHashCode;
import lombok.Getter;
import lombok.NonNull;
import lombok.ToString;
import lombok.experimental.SuperBuilder;
import okhttp3.HttpUrl;

/**
 * The request to retrieve a paginated list of collection items for a release.
 *
 * @see PaginatedRequestBase
 */
@SuperBuilder
@Getter
@EqualsAndHashCode(callSuper = true)
@ToString(callSuper = true)
public class GetCollectionItemsByReleaseRequest extends PaginatedRequestBase {
    /** The username (required). */
    @PathParameter
    private final String username;
    /** The release identifier (required). */
    @PathParameter
    private final long releaseId;

    @Override
    public HttpUrl.Builder populateQueryParameters(@NonNull final HttpUrl.Builder urlBuilder) {
        Validate.notBlank(username, "username must not be blank");
        Validate.isTrue(releaseId > 0L, "releaseId must be > 0");
        return super.populateQueryParameters(urlBuilder);
    }
}
