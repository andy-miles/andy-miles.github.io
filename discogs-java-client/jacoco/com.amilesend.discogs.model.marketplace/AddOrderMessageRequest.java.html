package com.amilesend.discogs.model.marketplace;

import com.amilesend.client.parse.strategy.GsonExclude;
import com.amilesend.client.util.Validate;
import com.amilesend.discogs.model.BodyBasedRequest;
import com.amilesend.discogs.model.BodyParameter;
import com.amilesend.discogs.model.PathParameter;
import lombok.Builder;
import lombok.Data;
import lombok.NonNull;
import okhttp3.HttpUrl;

import java.util.Objects;

/**
 * Describes a request to add a message to an order. Note: either {@link #message} or {@link #status} must
 * be defined.
 *
 * @see BodyBasedRequest
 */
@Builder
@Data
public class AddOrderMessageRequest implements BodyBasedRequest {
    /** The order identifier (required). */
    @PathParameter
    @GsonExclude
    private final String orderId;
    /** The message (optional). */
    @BodyParameter
    private final String message;
    /** The status (optional). */
    @BodyParameter
    private final String status;

    @Override
    public HttpUrl.Builder populateQueryParameters(@NonNull final HttpUrl.Builder urlBuilder) {
        Validate.notBlank(orderId, "orderId must not be blank");
        Validate.isTrue(Objects.nonNull(message) || Objects.nonNull(status),
                "Either message or status must be defined");
        return urlBuilder;
    }
}
