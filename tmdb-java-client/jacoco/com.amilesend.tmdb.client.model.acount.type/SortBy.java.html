package com.amilesend.tmdb.client.model.acount.type;

import lombok.Getter;
import lombok.RequiredArgsConstructor;

/** Defines how results are sorted. */
@RequiredArgsConstructor
@Getter
public enum SortBy {
    /** Sort results by the created at timestamp in ascending order. */
    CREATED_ASC("created_at.asc"),
    /** Sort results by the created at timesamp in descending order. */
    CREATED_DSC("created_at.desc");

    /** The associated query parameter value. */
    private final String value;

    @Override
    public String toString() {
        return value;
    }
}
