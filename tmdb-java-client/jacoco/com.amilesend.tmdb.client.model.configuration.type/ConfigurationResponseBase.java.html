package com.amilesend.tmdb.client.model.configuration.type;

import lombok.Data;

import java.util.List;

/**
 * Base class for response that pertain to configuration.
 *
 * @param <T> the type of results
 */
@Data
public abstract class ConfigurationResponseBase<T> {
    /** The list of results. */
    final List<T> results;

    /**
     * Creates a new {@code ConfigurationResponseBase}.
     *
     * @param results the list of results
     */
    public ConfigurationResponseBase(final List<T> results) {
        this.results = results;
    }
}
